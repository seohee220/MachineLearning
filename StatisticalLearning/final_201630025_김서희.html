def feature_selection(k):
    new_train = SelectKBest(chi2, k=k).fit_transform(train_dat, train_label)
    selected_features = SelectKBest(chi2, k=k).fit(train_dat, train_label).get_support().tolist()
    selected_features_idx =  [i for i,ft in enumerate(selected_features) if ft is True]
    new_cv_dat = cv_dat[selected_features_idx]
    new_test_dat = test_dat[selected_features_idx]
    return new_train, new_cv_dat, new_test_dat, selected_features_idx

def plot_mse_by_featureNum(mes_all, selected_features_idx_all, title):
    if len(mes_all)<13:
        len_x = range(3,14)
    else:
        len_x = range(1,14)
    plt.plot(len_x, mes_all)
    plt.xlabel('number of featues')
    plt.ylabel('TEST MSE')
    plt.title(title+' feature scailing')
    plt.show()
    
    min_mse =  min(mes_all)
    right_idx = mes_all.index(min(mes_all))
    right_k = right_idx+1
    right_features = selected_features_idx_all[right_idx]
    print 'minimum mse : ', min_mse
    print '성능이 가장 좋은 k : ', right_k
    print '선택된 k개 feature : ',right_features
    return right_k

def calMSE(test_y, predict_y):
    n = float(len(test_y))
    return sum((predict_y-test_y)**2)/n

def cal_sensiti_specifi(conf_mat):
    r0p0 = conf_mat[0][0]
    r0p1 = conf_mat[0][1]
    r1p0 = conf_mat[1][0]
    r1p1 = conf_mat[1][1]
    
    # sensitivity : AHD 1일 사람을 1이라 예측하는 비율
    sensitivity = 1-r1p0/float(r1p0+r1p1)
    # specificity : AHD 0일 사람을 1이라 예측하는 비율
    specificity = 1-r0p1/float(r0p0+r0p1)
    
    return str(sensitivity*100)+'%', str(specificity*100)+'%'

def training_knn():
    mse_all=[]; y_pred_all=[]; selected_features_idx_all=[]
    for n_features in range(1,14):
        new_train, new_cv_dat, new_test_dat, selected_features_idx = feature_selection(n_features)
        
        MSE_by_k=[]; y_pred=[]
        n_test = len(cv_label)
        for i in range(1,n_test+1):
            n_neighbors = i
            knn = neighbors.KNeighborsClassifier(n_neighbors=n_neighbors)
            knn_y_pred = knn.fit(new_train, train_label).predict(new_cv_dat)
            
            MSE_by_k.append(calMSE(cv_label, knn_y_pred))
            y_pred.append(knn_y_pred)
            
        mse_all.append(MSE_by_k)
        y_pred_all.append(y_pred)
        selected_features_idx_all.append(selected_features_idx)
    return mse_all, y_pred_all, selected_features_idx_all
