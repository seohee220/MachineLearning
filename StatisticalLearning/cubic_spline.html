def h(x,keusi):
    return array([max((xi-(keusi))**3, 0) for xi in x])

def func(x, b0, b1, b2, b3, b4, b5, b6):
    x_qua = x**2
    x_cub = x**3
    h1 = h(x,-pi)
    h2 = h(x,0)
    h3 = h(x,pi)
    return b0 + b1*x + b2*x_qua + b3*x_cub + b4*h1 +b5*h2 + b6*h3
